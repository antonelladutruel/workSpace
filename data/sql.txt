	`APELLIDO` VARCHAR(50) NOT NULL COLLATE 'latin1_swedish_ci',
	`SEGUNDO_APELLIDO` VARCHAR(50) NOT NULL COLLATE 'latin1_swedish_ci',
	`TELÃ‰FONO` VARCHAR(15) NOT NULL COLLATE 'latin1_swedish_ci',
	`EMAIL` VARCHAR(50) NOT NULL COLLATE 'latin1_swedish_ci',

;


CREATE TABLE `carrito` (
	`ID` INT(11) NOT NULL AUTO_INCREMENT,
	`NOMBRE` VARCHAR(50) NOT NULL COLLATE 'latin1_swedish_ci',
	`DESCRIPCION` VARCHAR(50) NOT NULL COLLATE 'latin1_swedish_ci',
	`PRECIO` DECIMAL(20,6) NOT NULL,
	`CANTIDAD_VENDIDOS` INT(11) NOT NULL,
	PRIMARY KEY (`ID`) USING BTREE
)
COLLATE='latin1_swedish_ci'
ENGINE=InnoDB
;
